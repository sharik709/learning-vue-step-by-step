HTML

<div id="app1" class="mb-3 bg-green-500 p-5 text-white">
    <h1>App 1</h1>
    <!-- following message is available because while instanting vue instance. We've provided data with message in it. -->
    <h1>{{ message }}</h1>
</div>

<div id="app2" class="mb-3 bg-green-500 p-5 text-white">
    <h1>App 2</h1>
    <!-- following message is available because while instanting vue instance. We've provided data with message in it. -->
    <h1>{{ message }}</h1>
</div>

JavaScript

       
        new Vue({
            el: '#app1',
            data: {
                message: 'Hello from VueJS Instance 1' 
            }
        })


        new Vue({
            el: '#app2',
            data: {
                message: 'Hello from VueJS Instance 2'
            }
        })
